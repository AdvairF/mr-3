    ```jsx
    const parcelasPorContrato = useMemo(() => {
      const m = new Map();
      (allDividas || []).forEach(d => {
        if (!d.contrato_id) return;
        const k = String(d.contrato_id);
        if (!m.has(k)) m.set(k, []);
        m.get(k).push(d);
      });
      return m;
    }, [allDividas]);
    ```
