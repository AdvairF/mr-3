```js
// ─── PAGAMENTOS POR CONTRATO (Phase 7) ───────────────────────────────────────

/**
 * Registra pagamento no nível do contrato via stored procedure.
 * SP amortiza parcelas abertas pela mais antiga (Art. 354 CC), insere em pagamentos_contrato,
 * registra evento 'pagamento_recebido' em contratos_historico.
 * Retorna { parcelas_amortizadas: N, parcelas_ids: [...] } (PAGCON-02, PAGCON-03, HIS-05).
 */
export async function registrarPagamentoContrato(contratoId, { data_pagamento, valor, observacao }) {
  return sb("rpc/registrar_pagamento_contrato", "POST", {
    p_contrato_id:    contratoId,
    p_data_pagamento: data_pagamento,
    p_valor:          valor,
    p_observacao:     observacao ?? null,
  });
}

/**
 * Reverte um pagamento registrado via stored procedure.
 * SP desfaz amortização das parcelas afetadas, remove row de pagamentos_contrato,
 * registra evento 'pagamento_revertido' em contratos_historico (PAGCON-06, HIS-05).
 */
export async function excluirPagamentoContrato(pagamentoId) {
  return sb("rpc/reverter_pagamento_contrato", "POST", {
    p_pagamento_id: pagamentoId,
  });
}

/**
 * Lista pagamentos registrados para um contrato em ordem cronológica.
 * PAGCON-04: usado pelo lazy-load da seção "Pagamentos Recebidos" no DetalheContrato.
 */
export async function listarPagamentosContrato(contratoId) {
  return dbGet(PAG_TABLE, `contrato_id=eq.${encodeURIComponent(contratoId)}&order=data_pagamento.asc`);
}
```
