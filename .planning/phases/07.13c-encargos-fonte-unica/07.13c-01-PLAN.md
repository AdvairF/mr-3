WITH dividas_contrato AS (
  SELECT d.*, cd.multa_percentual AS contrato_multa,
         cd.juros_am_percentual AS contrato_juros,
         cd.honorarios_percentual AS contrato_honor,
         cd.indice_correcao AS contrato_indice,
         cd.juros_tipo AS contrato_juros_tipo,
         cd.art523_opcao AS contrato_523
  FROM dividas d
  JOIN contratos_dividas cd ON cd.id = d.contrato_id
  WHERE d.contrato_id = '335a2ad2-9481-4836-a88a-55fbe6375827'
),
calc_antes AS (
  SELECT
    SUM(d.valor_total) AS principal_antes,
    -- correção, juros, multa, honor calculados aplicando d.*_percentual e d.indice_correcao
    -- (fórmulas dependem de funções SQL helper — orchestrator implementa funções inline ou usa CTE)
    SUM(...) AS correcao_antes, SUM(...) AS juros_antes, SUM(...) AS multa_antes,
    SUM(...) AS honor_antes
  FROM dividas_contrato d
),
calc_depois AS (
  SELECT
    SUM(d.valor_total) AS principal_depois,
    -- mesma estrutura mas usando d.contrato_multa, d.contrato_juros, d.contrato_honor, d.contrato_indice, d.contrato_juros_tipo
    SUM(...) AS correcao_depois, SUM(...) AS juros_depois, SUM(...) AS multa_depois,
    SUM(...) AS honor_depois
  FROM dividas_contrato d
)
SELECT
  '335a2ad2-9481-4836-a88a-55fbe6375827' AS contrato_id,
  a.principal_antes, d.principal_depois, (d.principal_depois - a.principal_antes) AS delta_principal,
  a.correcao_antes, d.correcao_depois, (d.correcao_depois - a.correcao_antes) AS delta_correcao,
  a.juros_antes, d.juros_depois, (d.juros_depois - a.juros_antes) AS delta_juros,
  a.multa_antes, d.multa_depois, (d.multa_depois - a.multa_antes) AS delta_multa,
  a.honor_antes, d.honor_depois, (d.honor_depois - a.honor_antes) AS delta_honor,
  (a.principal_antes + a.correcao_antes + a.juros_antes + a.multa_antes + a.honor_antes) AS total_antes,
  (d.principal_depois + d.correcao_depois + d.juros_depois + d.multa_depois + d.honor_depois) AS total_depois
FROM calc_antes a, calc_depois d;
